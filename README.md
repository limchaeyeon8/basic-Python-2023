```python

#문자열 포맷팅
pi = 3.1415928979

print(f'파이는 {pi:0.2f}입니다.')       #파이는 3.14입니다. # 소수점 2번째 자리까지 출력
print(f'파이는 {pi:10.3f}입니다.', end = '\n\n')       #파이는      3.142입니다.
                                               # 앞에 공백 + 소수점 3번째 자리까지 출력
```
